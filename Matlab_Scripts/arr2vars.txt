% Copy and paste this code into MATLAB
% Adjust the unit variable to whichever unit you're working with
unit = 1;
clear -regexp ^adch_;
for st = 1 : data{unit}.NumUnits
    unitVarStr = data{unit}.Names{st};
    spikesStr= ['data{' num2str(unit) '}.Spikes{' num2str(st) '}'];
    eval([unitVarStr '=' spikesStr ';']);
end;
clear('st', 'unitVarStr', 'spikesStr', 'unit', 'ans');
